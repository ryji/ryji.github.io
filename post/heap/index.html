Heap implement in C#

2018-10-02
C# Date Structure

Contents

堆结构是一种逻辑结构为完全二叉树的数据结构，且每一级的父节点和子节点均满足一定的大小关系，同级节点的大小关系不作要求。对于最大/最小元素的查找，堆结构的时间复杂度为O(1)， 其他元素的查找时间复杂度为O(N) ，元素的插入和删除的时间复杂度为O(lgN)。

代码实现

/// <summary>
/// max/min heap
/// </summary>
/// <typeparam name="T"></typeparam>
public class Heap<T>
{
    /// <summary>
    /// elemet count in Binary Heap
    /// </summary>
    public int Count { get; set; }

    private Func<T, T, bool> _compare;

    /// <summary>
    /// for max-heap, Compare return true when param1 > param2
    /// for min-heap, Compare return true when param1 < param2
    /// </summary>
    public Func<T, T, bool> Compare
    {
        get { return _compare; }
        set
        {
            Debug.Assert(_compare == null, "Compare can only be set once");
            _compare = value;
        }
    }


    private Func<T, T, bool> _equals;


    public Func<T,T,bool> Equals
    {
        get { return _equals; }
        set
        {
            Debug.Assert(_equals == null, "Equals can only be set once");
            _equals = value;
        }
    }

    private List<T> _list;
    /// <summary>
    /// constructor
    /// </summary>
    public Heap() : this(null)
    {

    }
    /// <summary>
    /// constructor
    /// </summary>
    /// <param name="compare"></param>
    public Heap(Func<T, T, bool> compare) : this(null, compare)
    {

    }

    /// <summary>
    /// constructor
    /// </summary>
    /// <param name="compare"></param>
    /// <param name="equals"></param>
    public Heap(Func<T, T, bool> compare, Func<T, T, bool> equals) : this(null, compare, equals)
    {

    }


    /// <summary>
    /// constructor
    /// </summary>
    /// <param name="items"></param>
    /// <param name="compare"></param>
    public Heap(IList<T> items, Func<T, T, bool> compare, Func<T, T, bool> equals)
    {
        Count = 0;
        Compare = compare;
        Equals = equals;
        _list = new List<T>();
        if (items == null) return;
        for (int i = 0; i < items.Count; ++i)
        {
            Insert(items[i]);
        }
    }


    /// <summary>
    /// find item equals with item, idx is item index in heap
    /// </summary>
    /// <param name="item"></param>
    /// <param name="idx"></param>
    /// <returns></returns>
    public T LookUp(T item, out int idx)
    {
        Debug.Assert(Compare != null, "Compare is null");
        Debug.Assert(item != null, "item is null");
        Debug.Assert(Equals != null, "Equals is null");

        for (int i = 0; i < _list.Count; ++i)
        {
            if (Equals(_list[i], item))
            {
                idx = i;
                return _list[i];
            }
        }
        idx = -1;
        return default(T);
    }

    /// <summary>
    /// insert an element into heap, if exists duplicate item, update it.
    /// </summary>
    /// <param name="item"></param>
    public void Insert(T item)
    {
        Debug.Assert(Compare != null, "Compare is null");
        Debug.Assert(item != null, "item is null");
        var idx = -1;
        LookUp(item, out idx);
        if (idx >= 0)
        {
            UpdateAt(idx, item);
        }
        else
        {
            _list.Add(item);
            ShiftUp(Count, item);
            Count++;
        }
    }

    /// <summary>
    /// move the item up in binary tree
    /// </summary>
    /// <param name="holeIdx"></param>
    /// <param name="item"></param>
    private void ShiftUp(int holeIdx, T item)
    {
        int pIdx = (holeIdx - 1) / 2;
        //target hole is either the root or Compare item and holeIdx's parent returns false
        while (holeIdx > 0 && Compare(item, _list[pIdx]))
        {
            _list[holeIdx] = _list[pIdx];
            holeIdx = pIdx;
            pIdx = (holeIdx - 1) / 2;
        }
        _list[holeIdx] = item;
    }

    /// <summary>
    /// check if all element in heap match Compare
    /// </summary>
    /// <returns></returns>
    public bool Check()
    {
        return CheckAt(0);
    }

    /// <summary>
    /// print the heap in order of depth of binary tree
    /// </summary>
    public void Print()
    {
        var lines = new List<string>();
        GetTreeLine(0, 0, lines);
        for (int i = 0; i < lines.Count; ++i)
        {
            Console.WriteLine(lines[i]);
        }

    }

    /// <summary>
    /// check element in idx and it's sub tree 
    /// </summary>
    /// <param name="idx"></param>
    /// <returns></returns>
    private bool CheckAt(int idx)
    {
        if (idx >= Count)
        {
            return true;
        }
        var pIdx = (idx - 1) / 2;
        var scheck = pIdx == idx || Compare(_list[pIdx], _list[idx]);
        var rc = 2 * (idx + 1);
        return scheck && CheckAt(rc - 1) && CheckAt(rc);
    }

    private void GetTreeLine(int idx, int depth, List<string> lines)
    {
        if (idx >= Count) return;
        if (depth == lines.Count)
        {
            lines.Add(string.Empty);
        }
        lines[depth] += string.Format("{0};", _list[idx]);
        GetTreeLine(2 * (idx + 1) - 1, depth + 1, lines);
        GetTreeLine(2 * (idx + 1), depth + 1, lines);
    }

    /// <summary>
    /// pop the max/min element 
    /// </summary>
    public void Pop()
    {
        if (Compare == null)
        {
            throw new ArgumentException("compare is null");
        }
        if (_list.Count == 0) return;
        var item = _list[Count - 1];
        Count--;
        ShiftDown(0, item);
        _list.RemoveAt(Count);
    }

    /// <summary>
    /// move an item down in binary tree
    /// </summary>
    /// <param name="holeIdx"></param>
    /// <param name="item"></param>
    private void ShiftDown(int holeIdx, T item)
    {
        var sc = 2 * (holeIdx + 1);
        //target holdIdx is either a leaf or Compare item with it's children both returns false 
        while (sc <= Count)
        {
            sc -= sc == Count || Compare(_list[sc - 1], _list[sc]) ? 1 : 0;
            if (Compare(_list[sc], item))
            {
                _list[holeIdx] = _list[sc];
                holeIdx = sc;
                sc = 2 * (holeIdx + 1);
            }
            else
            {
                break;
            }
        }
        _list[holeIdx] = item;
    }

    /// <summary>
    /// peek the max/min element in heap.
    /// when heap in empty, return default(T)
    /// </summary>
    /// <returns></returns>
    public T Peek()
    {
        if (Count == 0)
        {
            return default(T);
        }
        return _list[0];
    }

    /// <summary>
    /// replace heap item at idx with item, do not consider if duplicate item exists when update
    /// </summary>
    /// <param name="idx"></param>
    /// <param name="item"></param>
    public void UpdateAt(int idx, T item)
    {
        Debug.Assert(Count >= 0);
        Debug.Assert(idx >= 0);
        if (Compare(_list[idx], item))
        {
            ShiftDown(idx, item);
        }
        else
        {
            ShiftUp(idx, item);
        }
    }

    /// <summary>
    /// delete item at idx
    /// </summary>
    /// <param name="idx"></param>
    public void DeleteAt(int idx)
    {
        Debug.Assert(Count > 0);
        Debug.Assert(idx >= 0);
        Count--;
        UpdateAt(idx, _list[Count]);
        _list.RemoveAt(Count);
    }

}

测试代码

void TestHeap()
{
    int repeat = 1000;
    int heapSize = 100;

    for (int i = 0; i < repeat; ++i)
    {
        Console.WriteLine("repeat:{0}", i);
        var heap = HeapInsertTest(heapSize);
        HeapPopTest(heap);
        heap = HeapInsertTest(heapSize);
        HeapUpdateTest(heap);
        heap = HeapInsertTest(heapSize);
        HeapDeleteTest(heap);
    }
}

Heap<int> HeapInsertTest(int heapSize)
{
    Console.WriteLine("HeapInsertTest");
    var hashset = new HashSet<int>();
    var random = new Random();
    while (hashset.Count != heapSize)
    {
        hashset.Add(random.Next(1, 10 * heapSize));
    }
    var heap = new Heap<int>((a, b) => a > b, (a, b) => a == b);
    foreach (var item in hashset)
    {
        heap.Insert(item);
        HeapCheck(heap);
    }
    return heap;
}

void HeapPopTest(Heap<int> heap)
{
    Console.WriteLine("HeapPopTest");
    while (heap.Count != 0)
    {
        heap.Pop();
        HeapCheck(heap);
    }
}

void HeapUpdateTest (Heap<int> heap)
{
    Console.WriteLine("HeapUpdateTest");
    for (int i = 0; i < heap.Count; ++i)
    {
        heap.UpdateAt(i, heap.Peek() + 1);
        HeapCheck(heap);
    }
}


void HeapDeleteTest (Heap<int> heap)
{
    Console.WriteLine("HeapDeleteTest");
    var random = new Random();
    var count = heap.Count;
    for (int i = 0; i < count; ++i)
    {
        var ranIdx = random.Next(0, heap.Count - 1);
        heap.DeleteAt(ranIdx);
        HeapCheck(heap);
    }
    
}

void HeapDeleLookUpTest(Heap<int> heap)
{
    Console.WriteLine("HeapDeleteTest");
    var random = new Random();
    var count = heap.Count;
    for (int i = 0; i < count; ++i)
    {
        var ranIdx = random.Next(0, heap.Count - 1);
        heap.DeleteAt(ranIdx);
        HeapCheck(heap);
    }
}

void HeapCheck<T>(Heap<T> heap)
{
    var res = heap.Check();
    if (!res)
    {
        heap.Print();
        Console.ReadLine();
    }
}
