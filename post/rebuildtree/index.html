rebuildTree

2018-05-30

Contents

        • 节点定义
        • 重建方法
        • 测试程序

最近遇到个需求：在数据库中有一组逻辑关系为树的记录，通过parentId和CurrentId来标记节点关系，需要重建这组记录的树状结构。如果通过节点Id一级一级查找子节点来重建树结构，在中间节点较多的情况下，会产生较多的数据库访问，影响系统效率。可以将数据库中的目标记录一次查出，直接重建树结构。

节点定义

public class TreeNode
{
    public int ParentId {get;set;}
    public int CurrentId {get;set;}
    public List<TreeNode> Items {get;set;}
    public bool IsLeaf {get;set;}
    public override string ToString()
    {
        return string.Format("ParentId:{0}, CurrentId{2}", ParentId, CurrentId);
    }
}

重建方法

先获取根节点， 获取父节点Id到其子节点列表的字典，最后递归添加子节点到父节点的Items属性。

public class TreeUtil
{
    public static TreeNode ReBuildTree(List<TreeNode> nodes)
    {
        var pCdict = new Dictionary<int, List<TreeNode>>();
        foreach(var node in nodes)
        {
            if (!pCdict.ContainsKey(node.ParentId))
            {
                pCdict[node.ParentId] = new List<TreeNode>();
            }
            pCdict[node.ParentId].Add(node);
        }
        var root = GetRoot(nodes);
        GetChilden(root, pCdict);
        return root;
    }

    private static void GetChilden(TreeNode node, Dictionary<int, List<TreeNode>> pCdict)
    {
        //到达叶子
        if (!pCdict.ContainsKey(node.CurrentId))
        {
            node.isLeaf = true;
            return;
        }
        node.Items = pCdict[node.CurrentId];
        node.isLeaf = false;
        foreach (var child in pCdict[node.CurrentId])
        {
            GetChilden(child, pCdict);
        }
    }

    //获取根节点，在数据库记录中没有根节点的情况下，重建一个根节点
    private static TreeNode GetRoot(List<TreeNode> nodes)
    {
        //子节点字典
        var cIds = new HashSet<int>();
        nodes.ForEach(item => cIds.Add(item.CurrentId));
        int count = 0;
        int index = -1;
        for (int i = 0; i < nodes.Count; ++i)
        {
            if (!cIds.Contains(nodes[i].ParentId)){
                count++;
                index = i;
            }
        }
        //只有一个父节点Id没有出现在子节点字典中，可以当做父节点使用
        if (count == 1) return nodes[index];
        var pId = cIds.Min() - 1;
        //新建一个根节点
        return new TreeNode() { ParentId = pId - 1, CurrentId = pId};
    }
}

public class Test
{

    public void TestRebuild()
    {
        //多个根节点
        var treeNodes = new List<TreeNode>()
        {
            new TreeNode(){ ParentId = 0, CurrentId = 1},
            new TreeNode(){ ParentId = 0, CurrentId = 2},
            new TreeNode(){ ParentId = 0, CurrentId = 3},
            new TreeNode(){ ParentId = 1, CurrentId = 4},
            new TreeNode(){ ParentId = 1, CurrentId = 5},
            new TreeNode(){ ParentId = 1, CurrentId = 6},
            new TreeNode(){ ParentId = 1, CurrentId = 7},
            new TreeNode(){ ParentId = 2, CurrentId = 8},
            new TreeNode(){ ParentId = 3, CurrentId = 9},
            new TreeNode(){ ParentId = 3, CurrentId = 10},
            new TreeNode(){ ParentId = 5, CurrentId = 11},
            new TreeNode(){ ParentId = 5, CurrentId = 12},
            new TreeNode(){ ParentId = 7, CurrentId = 13},
            new TreeNode(){ ParentId = 11, CurrentId = 14},
            new TreeNode(){ ParentId = 3, CurrentId = 15},
        };

        var root = TreeUtil.ReBuildTree(treeNodes);
        PrintTree(root);
        Console.WriteLine("-------------------------------------------------");
        //一个根节点
        var treeNodes2 = new List<TreeNode>()
        {
            new TreeNode(){ ParentId = 0, CurrentId = 1},
            new TreeNode(){ ParentId = 1, CurrentId = 2},
            new TreeNode(){ ParentId = 1, CurrentId = 3},
            new TreeNode(){ ParentId = 1, CurrentId = 4},
            new TreeNode(){ ParentId = 1, CurrentId = 5},
            new TreeNode(){ ParentId = 1, CurrentId = 6},
            new TreeNode(){ ParentId = 1, CurrentId = 7},
            new TreeNode(){ ParentId = 2, CurrentId = 8},
            new TreeNode(){ ParentId = 3, CurrentId = 9},
            new TreeNode(){ ParentId = 3, CurrentId = 10},
            new TreeNode(){ ParentId = 5, CurrentId = 11},
            new TreeNode(){ ParentId = 5, CurrentId = 12},
            new TreeNode(){ ParentId = 7, CurrentId = 13},
            new TreeNode(){ ParentId = 11, CurrentId = 14},
            new TreeNode(){ ParentId = 3, CurrentId = 15},
        };
        root = TreeUtil.ReBuildTree(treeNodes2);
        PrintTree(root);
    }

    private void PrintTree(TreeNode node)
    {
        Console.WriteLine(node.ToString());
        if (node.Items == null) return;
        foreach(var child in node.Items)
        {
            PrintTree(child);
        }
    }
}
